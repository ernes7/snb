{# Inline SVG sparkline — self-normalized to min/max of its own series. Empty/single-point series render a flat line. #} {% macro sparkline(points, width=60, height=18) %} {% if points and points|length > 1 %} {% set vmin = points|min %} {% set vmax = points|max %} {% set span = vmax - vmin %} {% set step = width / (points|length - 1) %} {% set coords = [] %} {% for v in points %} {% set x = (loop.index0 * step)|round(2) %} {% if span > 0 %} {% set y = (height - ((v - vmin) / span) * height)|round(2) %} {% else %} {% set y = (height / 2)|round(2) %} {% endif %} {% set _ = coords.append(x ~ ',' ~ y) %} {% endfor %} {% else %} {% endif %} {% endmacro %}
